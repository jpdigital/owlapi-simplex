27  import org.semanticweb.owlapi.util.BidirectionalShortFormProvider;
28  import org.semanticweb.owlapi.util.BidirectionalShortFormProviderAdapter;
29  import org.semanticweb.owlapi.util.ShortFormProvider;
30  import org.semanticweb.owlapi.util.SimpleShortFormProvider;

39  /**
40   * An Utility class for working with OWL individuals.
41   *
42   * @author <a href="mailto:jens.pelzetter@googlemail.com">Jens Pelzetter</a>
43   */
44  public class Instances {
45  
46      /**
47       * The ontology to use.
48       */
49      private final OWLOntology ontology;
50  
51      /**
52       * The ontology manager for interacting the ontology.
53       */
54      private final OWLOntologyManager ontologyManager;
55  
56      /**
57       * Reasoner for the ontology.
58       */
59      private final OWLReasoner reasoner;
60  
61      /**
62       * Creates a new instance.
63       *
64       * @param ontology        The ontology to use.
65       * @param ontologyManager The ontology manager.
66       * @param reasoner        The reasoner.
67       */
68      private Instances(
69          final OWLOntology ontology,
70          final OWLOntologyManager ontologyManager,
71          final OWLReasoner reasoner
72      ) {
73          this.ontology = ontology;
74          this.ontologyManager = ontologyManager;
75          this.reasoner = reasoner;
76      }
77  
78      /**
79       * Factory method for generating new intances of {
80       *
81       * @ocde DataProperties}.
82       *
83       * @param ontology        The ontology to use.
84       * @param ontologyManager The ontology manager.
85       * @param reasoner        The reasoner.
86       *
87       * @return A {@code DataProperties} instance.
88       */
89      public static Instances buildInstances(
90          final OWLOntology ontology,
91          final OWLOntologyManager ontologyManager,
92          final OWLReasoner reasoner
93      ) {
94          Objects.requireNonNull(ontology);
95          Objects.requireNonNull(ontologyManager);
96          Objects.requireNonNull(reasoner);
97  
98          return new Instances(ontology, ontologyManager, reasoner);
99      }
100 
101     /**
102      * Retrieves the instances of an OWL class.
103      *
104      * @param query The query describing the OWL class.
105      *
106      * @return All instances matching the DL query.
107      */
108     public Stream<OWLNamedIndividual> instances(final String query) {
109 
110         final ShortFormProvider shortFormProvider
111                                     = new SimpleShortFormProvider();
112         final Set<OWLOntology> importsClosure = ontology
113             .importsClosure()
114             .collect(Collectors.toSet());
115 
116         final BidirectionalShortFormProvider bidiShortformProvider
117                                                  = new BidirectionalShortFormProviderAdapter(
118                 ontologyManager,
119                 importsClosure,
120                 shortFormProvider);
121 
122         final ManchesterOWLSyntaxParser parser = OWLManager
123             .createManchesterParser();
124         parser.setDefaultOntology(ontology);
125         parser.setOWLEntityChecker(
126             new ShortFormEntityChecker(bidiShortformProvider));
127 
128         final OWLClassExpression queryExpression = parser
129             .parseClassExpression(query);
130 
131         return reasoner
132             .getInstances(queryExpression, false)
133             .entities()
134             .sorted(this::sortResults);
135     }
136 
137     /**
138      * Find all instances matching the provided DL query.
139      * 
140      * @param query The query to execute.
141      * @return All indviduals matching the query.
142      */
143     public List<OWLNamedIndividual> getInstances(final String query) {
144         return instances(query).collect(Collectors.toList());
145     }
146 
147     /**
148      * Get all instances with the provided IRI.
149      * 
150      * @param iri The IRI.
151      * @return A stream with all matching individuals.
152      */
153     public Stream<OWLNamedIndividual> instances(final IRI iri) {
154         return instances(iri.getShortForm());
155     }
156 
157     /**
158      * Get all instances with the provided IRI.
159      * 
160      * @param iri The IRI.
161      * @return A list with all matching individuals.
162      */
163     public List<OWLNamedIndividual> getInstances(final IRI iri) {
164         return instances(iri).collect(Collectors.toList());
165     }
166 
167     /**
168      * Helper method for sorting {@link OWLNamedIndividual}s by their IRI.
169      *
170      * @param result1
171      * @param result2
172      *
173      * @return
174      */
175     private int sortResults(final OWLNamedIndividual result1,
176                             final OWLNamedIndividual result2) {
177 
178         return result1
179             .getIRI()
180             .toString()
181             .compareTo(result2.getIRI().toString());
182     }
183 
184 }
