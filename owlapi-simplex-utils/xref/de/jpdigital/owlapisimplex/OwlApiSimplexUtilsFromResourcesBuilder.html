17  package de.jpdigital.owlapisimplex;
18  
19  import org.semanticweb.owlapi.model.OWLOntology;
20  import org.semanticweb.owlapi.model.OWLOntologyCreationException;
21  
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  /**
28   * An implementation of {@link OwlApiSimplexUtilsBuilder} which loads the 
29   * ontology document from the class path.
30   * 
31   * @author <a href="mailto:jens.pelzetter@googlemail.com">Jens Pelzetter</a>
32   */
33  public class OwlApiSimplexUtilsFromResourcesBuilder
34      extends OwlApiSimplexUtilsBuilder {
35  
36      /**
37       * The paths of the resources to load.
38       */
39      private final List<String> resourcePaths;
40  
41      public OwlApiSimplexUtilsFromResourcesBuilder(
42          final List<String> resourcePaths
43      ) {
44          this.resourcePaths = resourcePaths;
45      }
46  
47      /**
48       * Load the ontology documents from the class path.
49       *
50       * @return A list of the loaded ontologies.
51       *
52       * @throws OwlApiSimplexException If an error ocurrs, for example if one of
53       *                                the provided resources is not readable.
54       */
55      @Override
56      public List<OWLOntology> loadOntologies() throws OwlApiSimplexException {
57          final List<OWLOntology> ontologies = new ArrayList<>();
58          for (final String resourcePath : resourcePaths) {
59              try ( InputStream inputStream = getClass().getResourceAsStream(
60                  resourcePath
61              )) {
62                  final OWLOntology ontology = loadOntology(inputStream);
63                  ontologies.add(ontology);
64              } catch (IOException | OWLOntologyCreationException ex) {
65                  throw new OwlApiSimplexException(
66                      String.format(
67                          "Failed to load ontology from OWL file resource %s",
68                          resourcePath
69                      ),
70                      ex
71                  );
72              }
73          }
74          
75          return ontologies;
76      }
77  
78  }
